 34 float randpval (void)
 35 {
 36     int vr = rand();
 37     int vm = rand()%vr;
 38     float r = ((float)vm)/(float)vr;
 39     assert(r>=0.0f && r<=1.00001f);

 89 void initializeThreadData(threadData_t * cur, int i, int threads,int n,float * mVec1,float * nVec1,float * LVec1,float * RVec1,float* CVec1,float* FVec1)
 90 {
 91     cur->threadID=i;
 92     cur->threadTOTAL=threads;
 93     cur->threadBARRIER=0;
 94     cur->threadOPERATION=BUSYWAIT;
 95 
 96     cur->begin=(n/threads)*i;     
 97     cur->end  =(n/threads)*(i+1);
 98 
 99     cur->mVec=mVec1;
100     cur->nVec=nVec1;
101     cur->LVec=LVec1;
102     cur->RVec=RVec1;
103     cur->CVec=CVec1;
104     cur->FVec=FVec1;
105 
106     cur->avgF= _mm_set_ps(0,0,0,0);
107     cur->maxF= _mm_set_ps(0,0,0,0);
108     cur->minF= _mm_set_ps(FLT_MAX,FLT_MAX,FLT_MAX,FLT_MAX);
109 }

123 void paralsin(threadData_t * threadData)
124 {
125 
126     __m128 variable,variable1,variable2,variable3,variable4,variable5,variable6;
127     __m128 scale1 = _mm_set_ps1(0.01f);
128     __m128 scale2 = _mm_set_ps1(1.0f);
129     __m128 scale3 = _mm_set_ps1(2.0f);
130 
131     int     i= threadData->begin ;
132     int     end= threadData->end ;    
133 
134     float * mVec=threadData->mVec;
135     float * nVec=threadData->nVec;
136     float * LVec=threadData->LVec;
137     float * RVec=threadData->RVec;
138     float * CVec=threadData->CVec;
139     float * FVec=threadData->FVec;
140     
141 
142     for(;i<end;i+=4)
143     {
144 
145         __m128 LVecss= _mm_set_ps(LVec[i+3], LVec[i+2], LVec[i+1], LVec[i]);
146         __m128 RVecss= _mm_set_ps(RVec[i+3], RVec[i+2], RVec[i+1], RVec[i]);
147         __m128 mVecss= _mm_set_ps(mVec[i+3], mVec[i+2], mVec[i+1], mVec[i]);
148         __m128 nVecss= _mm_set_ps(nVec[i+3], nVec[i+2], nVec[i+1], nVec[i]);
149         __m128 CVecss= _mm_set_ps(CVec[i+3], CVec[i+2], CVec[i+1], CVec[i]);
150         __m128 FVecss= _mm_set_ps(FVec[i+3], FVec[i+2], FVec[i+1], FVec[i]);
151 
152         variable= _mm_add_ps(LVecss, RVecss);
153         variable1= _mm_div_ps( _mm_mul_ps(mVecss, _mm_sub_ps(mVecss,scale2))  ,  scale3);
154         variable2= _mm_div_ps( _mm_mul_ps(nVecss, _mm_sub_ps(nVecss,scale2))  ,  scale3);
155         variable3= _mm_div_ps(variable,_mm_add_ps(variable1,variable2));
156         variable4=_mm_sub_ps(CVecss,_mm_sub_ps(LVecss,RVecss));
157         variable5=_mm_mul_ps(mVecss,nVecss);
158         variable6=_mm_div_ps(variable4,variable5);
159 
160         FVecss = _mm_div_ps(variable3, _mm_add_ps(variable6, scale1));
161 
162         threadData->avgF= _mm_add_ps(FVecss,threadData->avgF);
163         threadData->maxF= _mm_max_ps(FVecss,threadData->maxF);
164         threadData->minF= _mm_min_ps(FVecss,threadData->minF);
165     }
166 }

168 void syncThreadsBARRIER(threadData_t * threadData)
169 {
170     threadData[0].threadOPERATION=BUSYWAIT;
171      pthread_barrier_wait(&barrier);
172 }

174 void setThreadOperation(threadData_t * threadData, int operation)
175 {
176     int i, threads=threadData[0].threadTOTAL;
177     
178     for(i=0;i<threads;i++)
179         threadData[i].threadOPERATION = operation;    
180 }

182 
183 void startThreadOperations(threadData_t * threadData, int operation)
184 {
185     setThreadOperation(threadData, operation);
186 
187     if(operation==LOOP)
188         paralsin(&threadData[0]);
189 
190     threadData[0].threadBARRIER=1;
191     syncThreadsBARRIER(threadData);        
192 }
193 
194 void terminateWorkerThreads(pthread_t * workerThreadL, threadData_t * threadData)
195 {
196     int i, threads=threadData[0].threadTOTAL;
197     
198     for(i=0;i<threads;i++)
199         threadData[i].threadOPERATION = EXIT;            
200 
201     for(i=1;i<threads;i++)
202         pthread_join(workerThreadL[i-1],NULL);
203 }

206 void * thread (void * x)
207 {
208     threadData_t * currentThread = (threadData_t *) x;
209 
210     int tid = currentThread->threadID;
211     int threads = currentThread->threadTOTAL;
212     
213     while(1)
214     {
215         __sync_synchronize();
216 
217         if(currentThread->threadOPERATION==EXIT)
218             return NULL;
219         
220         if(currentThread->threadOPERATION==LOOP)
221         {
222             paralsin (currentThread);
223             currentThread->threadOPERATION=BUSYWAIT;
224              pthread_barrier_wait(&barrier);        
225         }
226     }
227     
228     return NULL;        
229 }

231 int main(int argc, char ** argv)
232 {    
233     assert(argc==2);
234 
235 
236     float avgF = 0.0f;
237     float maxF = 0.0f;
238     float minF = FLT_MAX;
239 
240 
241     double timeTotalMainStart = gettime();
242     unsigned int N = (unsigned int)atoi(argv[1]);
243     unsigned int iters = 10;
244     srand(1);
245     float * mVec = (float*)_mm_malloc(sizeof(float)*N,16);
246     assert(mVec!=NULL);
247     float * nVec = (float*)_mm_malloc(sizeof(float)*N,16);
248     assert(nVec!=NULL);
249     float * LVec = (float*)_mm_malloc(sizeof(float)*N,16);
250     assert(LVec!=NULL);
251     float * RVec = (float*)_mm_malloc(sizeof(float)*N,16);
252     assert(RVec!=NULL);
253     float * CVec = (float*)_mm_malloc(sizeof(float)*N,16);
254     assert(CVec!=NULL);
255     float * FVec = (float*)_mm_malloc(sizeof(float)*N,16);
256     assert(FVec!=NULL);
257 
258 
259     ////////////////////////////////////////////////////////////
260     //Threads
261     int threads = THREADS;    //We pass the number of threads 
262 
263     //We are going to use Pthread-Barrier as our choise of synchronization
264     int s = pthread_barrier_init(&barrier,NULL,(unsigned int)threads);    //We initialize the barrier
265     assert(s == 0);
266     
267 
268     workerThread=NULL;
269     workerThread=(pthread_t *) malloc (sizeof(pthread_t)*((unsigned long)threads-1)); //We write the workerThread as thread-1 cause the one thread is the master
270 
271     threadData_t * threadData = (threadData_t *) malloc (sizeof(threadData_t)*((unsigned long)threads));
272     assert(threadData!=NULL);
273 
274     for (int i = 0; i < threads; i++)
275     {
276         initializeThreadData(&threadData[i],i,threads,N,mVec,nVec,LVec,RVec,CVec,FVec);
277     }
278 
279     for (int i = 1; i < threads; i++)
280     {
281         pthread_create(&workerThread[i-1],NULL,thread,(void*)(&threadData[i]));
282     }
283     ////////////////////////////////////////////////////////////
284 
285 
286 
287     for(unsigned int i=0;i<N;i++)
288     {
289         mVec[i] = (float)(MINSNPS_B+rand()%MAXSNPS_E);
290         nVec[i] = (float)(MINSNPS_B+rand()%MAXSNPS_E);
291         LVec[i] = randpval()*mVec[i];
292         RVec[i] = randpval()*nVec[i];
293         CVec[i] = randpval()*mVec[i]*nVec[i];
294         FVec[i] = 0.0;
295 
296         assert(mVec[i]>=MINSNPS_B && mVec[i]<=(MINSNPS_B+MAXSNPS_E));
297         assert(nVec[i]>=MINSNPS_B && nVec[i]<=(MINSNPS_B+MAXSNPS_E));
298         assert(LVec[i]>=0.0f && LVec[i]<=1.0f*mVec[i]);
299         assert(RVec[i]>=0.0f && RVec[i]<=1.0f*nVec[i]);
300         assert(CVec[i]>=0.0f && CVec[i]<=1.0f*mVec[i]*nVec[i]);
301     }
302 
303     double timeOmegaTotalStart = gettime();
304 
305     for(unsigned int j=0;j<iters;j++)
306     {
307         updateThreadMMA(threadData);
308         startThreadOperations(threadData, LOOP);        
309     }
310 
311     float maxl[4];
312     float minl[4];
313     float suml[4];
314 
315     
316 
317     for (unsigned int k=0;k<threads;k++)
318     {
319         _mm_store_ps(maxl, (&threadData[k])->maxF);
320         _mm_store_ps(minl, (&threadData[k])->minF);
321         _mm_store_ps(suml, (&threadData[k])->avgF);
322         maxF = max(max(max(max(maxl[0], maxl[1]), maxl[2]), maxl[3]),maxF);
323         minF = min(min(min(min(minl[0], minl[1]), minl[2]), minl[3]),minF);
324         avgF = sum(sum(sum(sum(suml[0], suml[1]), suml[2]), suml[3]),avgF);
325         
326     }
327 
328     double timeOmegaTotal = gettime()-timeOmegaTotalStart;
329     double timeTotalMainStop = gettime();
330 
331     printf("Omega time %fs - Total time %fs - Min %e - Max %e - Avg %e\n",
332     timeOmegaTotal/iters, timeTotalMainStop-timeTotalMainStart, (double)minF, (double)maxF,(double)avgF/N);
333 
334     terminateWorkerThreads(workerThread,threadData);
335     pthread_barrier_destroy(&barrier);
336 
337     _mm_free(mVec);
338     _mm_free(nVec);
339     _mm_free(LVec);
340     _mm_free(RVec);
341     _mm_free(CVec);
342     _mm_free(FVec);
343 }
